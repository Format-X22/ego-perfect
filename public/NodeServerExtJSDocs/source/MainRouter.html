/**
 * Главный роутер приложения.
 */
Ext.define('B.MainRouter', {
    extend: 'B.AbstractRouter',

    requires: [],

    delegate: {},

    config: {
        callback: null
    },

    constructor: function () {
        this.callParent(arguments);
        B.Main.getExpressApp().use(this.getExpressRouter());
        this.getCallback()();
    }
});
